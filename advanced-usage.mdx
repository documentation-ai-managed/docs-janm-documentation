## Custom Integrations

Build custom integrations using Janm's webhook system and API to connect with proprietary systems or implement specialized workflows.

<Tabs>
  <Tab title="Webhook Handlers" icon="webhook">
    Create custom webhook handlers to process events and trigger actions in your systems.

    ```javascript
app.post('/webhooks/janm', (req, res) => {
  const { event, data } = req.body;
  
  switch (event) {
    case 'document.updated':
      // Update external system
      updateExternalSystem(data.document);
      break;
    case 'project.created':
      // Create corresponding project
      createProjectMapping(data.project);
      break;
  }
  
  res.status(200).send('OK');
});
```
  </Tab>
  <Tab title="API Clients" icon="code">
    Build custom API clients for languages not supported by official SDKs.

    <CodeGroup tabs="Ruby,PHP">
      ```ruby
require 'net/http'
require 'json'

class JanmClient
  def initialize(api_key)
    @api_key = api_key
    @base_url = 'https://api.janm.com/v1'
  end
  
  def get_projects
    uri = URI("#{@base_url}/projects")
    http = Net::HTTP.new(uri.host, uri.port)
    http.use_ssl = true
    
    request = Net::HTTP::Get.new(uri)
    request['Authorization'] = "Bearer #{@api_key}"
    
    response = http.request(request)
    JSON.parse(response.body)
  end
end
```
      ```php
<?php

class JanmClient {
    private $apiKey;
    private $baseUrl = 'https://api.janm.com/v1';
    
    public function __construct($apiKey) {
        $this->apiKey = $apiKey;
    }
    
    public function getProjects() {
        $ch = curl_init();
        curl_setopt($ch, CURLOPT_URL, $this->baseUrl . '/projects');
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
        curl_setopt($ch, CURLOPT_HTTPHEADER, [
            'Authorization: Bearer ' . $this->apiKey
        ]);
        
        $response = curl_exec($ch);
        curl_close($ch);
        
        return json_decode($response, true);
    }
}
?>
```
